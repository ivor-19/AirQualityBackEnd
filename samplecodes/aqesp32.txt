#include <WiFi.h>
#include <HTTPClient.h>
#include <DHT.h>

#define LEDPIN_GREEN 32
#define LEDPIN_BLUE 33

#define DHTPIN 17
#define DHTTYPE DHT22

DHT dht(DHTPIN, DHTTYPE);

// Replace with your network credentials
const char *ssid = "Wokwi-GUEST";
const char *password = "";

// Vercel API endpoint URL
const String apiUrl = "https://air-quality-back-end-v2.vercel.app/esp/updateEsp";

void setup() {
  Serial.begin(115200);
  Serial.print("Connecting to WiFi");
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) {
    delay(100);
    Serial.print(".");
  }
  Serial.println("Connected to WiFi");
  dht.begin();
  Serial.println("Hello, ESP32!");
  pinMode(LEDPIN_GREEN, OUTPUT);
  pinMode(LEDPIN_BLUE, OUTPUT);
  delay(10);
}

void loop() {
  // put your main code here, to run repeatedly:
  float temperature =  random(200, 300) / 10.0;
  float humidity = dht.readHumidity();

  Serial.println("Reading...");
  digitalWrite(LEDPIN_BLUE, HIGH);

  if (isnan(humidity) || isnan(temperature)) {
    Serial.println("Failed to read from DHT sensor!");
  } else {
    sendDataToAPI(temperature, humidity);
    digitalWrite(LEDPIN_GREEN, HIGH);
    Serial.println("Updated Successfully...");
    Serial.print("Temperature: ");
    Serial.print(temperature);
    Serial.print(" Â°C  Humidity (from sensor): ");
    Serial.print(humidity);
    Serial.println(" %");
  }

  delay(1000);
  digitalWrite(LEDPIN_GREEN, LOW);
  
}

void sendDataToAPI(float temperature, float humidity) {
  HTTPClient http;

  // Start HTTP request
  http.begin(apiUrl);

  // Set the content type to JSON
  http.addHeader("Content-Type", "application/json");

  // Create a JSON body
  String jsonBody = "{\"temperature\":" + String(temperature) + ",\"humidity\":" + String(humidity) + "}";

  // Send HTTP POST request
  int httpResponseCode = http.POST(jsonBody);

  // Check the response
  if (httpResponseCode > 0) {
    Serial.print("HTTP Response code: ");
    Serial.println(httpResponseCode);
  } else {
    Serial.print("Error code: ");
    Serial.println(httpResponseCode);
  }

  // Close the connection
  http.end();
}
